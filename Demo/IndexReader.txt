var writer = new IndexWriter(fsdDir, indexConfig);

//方法一
DirectoryReader reader = writer.GetReader(true);
if (!reader.IsCurrent())
{
	reader.Dispose();
	reader = writer.GetReader(true);
}

//方法二
var newReader = DirectoryReader.OpenIfChanged(reader);
if (newReader != null)
{
	reader.Dispose();
	reader = newReader;
}
